function findMindShare(who,what){
//  var whoNum=gsGetCount(who);
  var whatNum=gsGetCount(what);
  var whoWhatNum=gsGetCount(who+" "+what);
  return makePercent(whoWhatNum / whatNum);
}
function makePercent(num){ // e.g., .0010 becomes .10; 
 var S= ""+(num*100); 
 var decPoint=S.indexOf(".");
 if(decPoint<0)return S;
 return S.substring(0,6+decPoint); // truncated, not rounded.
}
